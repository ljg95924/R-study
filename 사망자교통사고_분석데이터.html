evaluation 
evaluation<-function(lst){
  accuracy<-sapply(lst,function(one_result){ # 
    return(sum(one_result$predicted==one_result$actual)
           /NROW(one_result$actual))
  })
  print(sprintf("MEAN +/- SD: %.3f +/- %.3f",
                mean(accuracy),sd(accuracy))) #, 
  return(accuracy)
}
